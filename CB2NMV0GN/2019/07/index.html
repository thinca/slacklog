package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())
	s := ""
	rs := []rune("ホイク")
	for _, i := range rand.Perm(2) {
		s += string(rs[i])
	}
	s += s + "の"
	rs = []rune("ホイモ")
	for _, i := range rand.Perm(2) {
		s += string(rs[i])
	}
	fmt.Println(s)
}
