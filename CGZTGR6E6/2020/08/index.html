const Obj = { name: "gorilla" } as const;
type ObjKey = keyof typeof Obj;
const ObjKeys = Object.keys(Obj) as readonly ObjKey[];

// 実装
ObjKeys.forEach(k => {
  console.log(obj[k])
});
