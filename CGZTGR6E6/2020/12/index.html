実際にやろうとしているのは hubot-vimexec で、裏で Vim にスクリプトを実行させる処理の部分です。
今色々手元で書き換えちゃってますが、相談当時のコードを簡略化して少しコメント足して載せますね。 __ で始まってる関数は簡略化で仮の適当な名前の関数に置き換えたやつです。

let finished = false;
const cancelPromise = new Promise<string>((_resolve, reject) => {
  const exitCallback = (exitCode: number): void => {
    finished = true;
    reject(new Error(`Vim exited: ${exitCode}`));
  };
  // 実行中に Vim が終了する場合がある。その場合は main で結果は得られない
  this.eventEmitter.on("exit", exitCallback);
  setTimeout(() => {
    this.eventEmitter.off("exit", exitCallback);
    if (!finished) {
      // timeout の場合は <C-c> を送ってスクリプトを停止させる
      __stop();
    }
    reject(new Error("Execution timeout"));
  }, this.timeoutMs);
});
const mainPromise = __makeMainPromise().then((result) => {
  finished = true;
  return result;
});
return Promise.race([mainPromise, cancelPromise]);
