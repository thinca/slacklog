Trump = Struct.new(:mark, :num)
all_cards = 
  %i[spade clover heart diamond].flat_map {|mark|
    (1..13).map {|num|
      Trump.new(mark, num)
    }
  }

def straight?(cards)
  sorted = cards.map(&:num).sort
  sorted.each_cons(2).map {|i, j| j - i }.all? {|x| x == 1 }
end

1000.times do
  card_set = all_cards.sample(5)
  pp card_set if straight?(card_set)
end
