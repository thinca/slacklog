こういう記事がありました。ちょっと変えて行番号とソースコードを得られることが確かめられました! https://qiita.com/PND/items/8b6f64c9b185724edc74
import sys

def func(x):
    a = x + 1
    b = a * 4
    return a + 2

def trace(frame, event, arg):
    if event == 'line':
        module = frame.f_globals.get('__name__')
        loader = frame.f_globals.get('__loader__')
        source = loader.get_source(module)
        lineno = frame.f_lineno
        line = source.splitlines()[lineno - 1]
        print(lineno, line)

    if event == 'return':
        print(frame.f_locals.items())

    return trace

sys.settrace(trace)
