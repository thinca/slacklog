}

	return nil
}

	for i := 0; i < b.N; i++ {
		o.methodFunc()
	}
}

	f := pureFunc
	b.ResetTimer()
	for i := 0; i < b.N; i++ {
		f()
	}
}

type methoder interface {
	methodFunc() error
}

func Benchmark_Interface(b *testing.B) {
	var m methoder = someObj{}
	b.ResetTimer()
	for i := 0; i < b.N; i++ {
		m.methodFunc()
	}
}

var m123 methoder = someObj{}

func Benchmark_FuncVar_Interface(b *testing.B) {

	}
