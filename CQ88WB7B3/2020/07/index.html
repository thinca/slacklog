function! s:buffers_delete_from_lines(lines) abort
  for line in a:lines
    let matches = matchlist(line, '^buffer \(\d\+\)$')
    if len(matches) >= 1
      execute 'Bdelete! ' . matches[1]
    else
      execute 'Bdelete! ' . line
    endif
  endfor
endfunction
