function! CopyHover()
  let l:hover = join(CocAction('getHover'))
  if empty(l:hover) | echoerr "No Hover!" | return | endif
  let @+ = l:hover
  let @* = l:hover
  echom "Hover copied!"
endfunction

nnoremap <silent> <leader>k :call CopyHover()<CR>

" Focus floating window with <C-w><C-w> {{{
if has('nvim')
  function! s:focus_floating() abort
    if !empty(nvim_win_get_config(win_getid()).relative)
      wincmd p
      return
    endif
    for winnr in range(1, winnr('$'))
      let winid = win_getid(winnr)
      let conf = nvim_win_get_config(winid)
      if conf.focusable && !empty(conf.relative)
        call win_gotoid(winid)
        return
      endif
    endfor
  endfunction
  nnoremap <silent> <C-w><C-w> :<C-u>call <SID>focus_floating()<CR>
endif
