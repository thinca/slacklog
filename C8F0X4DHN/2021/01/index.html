ゴリラ（筋トレはじめました） 22日 08:48:39 Slack
> 整理すると、*[[PromiseIsHandled]]がfalseのPromise*とは、*まだthenやcatch, finallyによってコールバック関数が登録されていないPromise*を指すことになります。
> ここまでで分かったことは、Promiseオブジェクトの[[PromiseIsHandled]]は、そのPromiseオブジェクトに対してthen, catch, finallyが呼ばれるか、またはそのPromiseオブジェクトがawaitされた場合にtrueになるフラグだということです。
> そして、このフラグがfalseである（＝まだthenなどが呼ばれていない）Promiseがrejectした（RejectPromise抽象操作が呼ばれた）場合にHostPromiseRejectionTrackerが呼ばれる、すなわちUnhandle Rejectionとして扱われるということです。

正常に catch されてるよう
messages clear

let s:V = vital#_vital#new()
let s:Promise = s:V.import('Async.Promise')

function! s:sh(...) abort
  echom 'sh' a:000
  return s:Promise.new({rv, rj -> rv('foo')})
endfunction

function! s:make_response(...) abort
  echom 'make_responsee' a:000
  " E492: Not an editor command
  bar
endfunction

function! s:make_error_responsee(...) abort
  echom 'make_error_responsee' a:000
endfunction

function! s:do(...) abort
  return call('s:sh', a:000)
        \.then(function('s:make_response', ['baz']))
        \.catch(function('s:make_error_responsee'))
endfunction

call s:do('qux')
