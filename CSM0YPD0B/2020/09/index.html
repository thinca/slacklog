#!/bin/sh

SHA=$(git log --format="%H" -n 1)
docker  build . --tag <http://ghcr.io/mattn/longcat:$SHA|ghcr.io/mattn/longcat:$SHA> --cache-from <http://ghcr.io/mattn/longcat:latest|ghcr.io/mattn/longcat:latest>
ID=$(docker images | head -n 2 | tail -n 1 | awk '{print $3}')
docker tag $ID <http://ghcr.io/mattn/longcat:latest|ghcr.io/mattn/longcat:latest>
docker push <http://ghcr.io/mattn/longcat:latest|ghcr.io/mattn/longcat:latest>
