function! s:open_pr() abort
  let file = expand('%s')
  let line = line('.')
  let blame_line = system(printf('git blame -L %s,%s -- %s', line, line, file))
  if empty(blame_line)
    return
  endif
  let id = blame_line->split(' ')[0]
  echom system(printf('%s %s', 'getpr', id))
endfunction
