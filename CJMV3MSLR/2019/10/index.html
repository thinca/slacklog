function! s:is_most_left(nr)
  let l:oldw = winnr()
  silent! exe "normal! \<c-w>h"
  let l:neww = winnr()
  silent! exe l:oldw.'wincmd w'
  return l:oldw == l:neww
endfunction

function! s:is_most_right(nr)
  let l:oldw = winnr()
  silent! exe "normal! \<c-w>l"
  let l:neww = winnr()
  silent! exe l:oldw.'wincmd w'
  return l:oldw == l:neww
endfunction

function! s:wider(c)
  let l:w = winnr()
  if s:is_most_left(l:w)
    exe 'normal! ' . a:c . "\<c-w><"
  elseif s:is_most_right(l:w)
    exe 'normal! ' . a:c . "\<c-w>>"
  endif
endfunction
